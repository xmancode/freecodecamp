 function generatePassword(pwdSize){
     let letters="ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()";
     let letterArry=[];
     letterArry=letters.split("");
     let result=[];
     
     for(let i=0;i<pwdSize;i++){
       let randIndex=Math.floor(Math.random()*letterArry.length);
            result.push(letterArry[randIndex]);
     }
     let password=result.join('');
     return password;
}

const password=generatePassword(7);
console.log(`Generated password: ${password}`);
